import pyexcel as pe

rows = pe.get_records(file_name="sales.xlsx")
summary = []
for row in rows:
    margin = row["Revenue"] - row["Cost"]
    summary.append({"Month": row["Month"], "Margin": margin})

pe.save_as(records=summary, dest_file_name="report.xlsx")
