from fastapi import APIRouter
from fastapi.responses import HTMLResponse

router = APIRouter()

@router.get("/", response_class=HTMLResponse)
def home():
    return render("index.html")

@router.get("/courses/{slug}")
def course(slug: str):
    return render(f"course-{slug}.html")
